Step 1: Identify Your Package Manager

Step 2: Refresh Package Index (Update Catalog)

# Ubuntu/Debian
sudo apt update

# Fedora/RHEL/CentOS
sudo dnf check-update

# Arch
sudo pacman -Sy

# openSUSE
sudo zypper refresh

# Alpine
sudo apk update

Step 3: Search for a Tool (example: htop)

# Ubuntu/Debian
apt search htop | head -n 20

# Fedora/RHEL/CentOS
dnf search htop | head -n 20

# Arch
pacman -Ss htop | head -n 20

# openSUSE
zypper search htop

# Alpine
apk search htop

Step 4: Install the Tool

# Ubuntu/Debian
sudo apt install -y htop

# Fedora/RHEL/CentOS
sudo dnf install -y htop

# Arch
sudo pacman -S --noconfirm htop

# openSUSE
sudo zypper install -y htop

# Alpine
sudo apk add htop

Step 5: Verify Installation

which htop
htop --version || htop -v

(Optional) Launch htop, then quit with q:

htop

Step 6: Remove / Uninstall

# Ubuntu/Debian
sudo apt remove -y htop
# (optional, purge config)
sudo apt purge -y htop
sudo apt autoremove -y

# Fedora/RHEL/CentOS
sudo dnf remove -y htop

# Arch
sudo pacman -R htop
# (or remove with deps no longer needed)
sudo pacman -Rns htop

# openSUSE
sudo zypper remove -y htop

# Alpine
sudo apk del htop

Step 7 (Optional): Install & Manage a Service (nginx)

# Install nginx
# Ubuntu/Debian
sudo apt install -y nginx
# Fedora/RHEL/CentOS
sudo dnf install -y nginx
# Arch
sudo pacman -S --noconfirm nginx
# openSUSE
sudo zypper install -y nginx
# Alpine
sudo apk add nginx

# Start & enable (systemd-based distros)
sudo systemctl start nginx
sudo systemctl enable nginx
sudo systemctl status nginx --no-pager

If your distro doesnâ€™t use systemd, try:

sudo service nginx start

Test locally (should return HTML):

curl -I http://127.0.0.1

Step 8 (Optional): List Installed Packages & Info

# Ubuntu/Debian
apt list --installed | head
apt show curl

# Fedora/RHEL/CentOS
dnf list installed | head
dnf info curl

# Arch
pacman -Q | head
pacman -Qi curl

# openSUSE
zypper se -i | head
zypper info curl

# Alpine
apk info | head
apk info -a curl
